 38 /**

 58 ct.send = function(url, data) {

 69 /**

 74  * @return {object} a list of all returned messages
 75  */
 76 ct.read = function(url, data) {

 82 /**
 83  * Look for all the active users in the last 5 minutes to list in the sidebar
 84  *

 89 ct.userCheck = function(url, data) {

 95 /**
 96  * Refresh the chat window at an interval
 97  * 
 98  * @param {string} a url representing a controller method server-side

102 ct.ref = function(url, data) {
103
